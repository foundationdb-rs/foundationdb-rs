20#[derive(Clone)]
21struct Counter {
22    size: usize,
23    inner: Arc<AtomicUsize>,
24}
25impl Counter {
26    fn new(size: usize) -> Self {
27        Self {
28            size,
29            inner: Default::default(),
30        }
31    }
32
33    fn decr(&self, n: usize) -> bool {
34        let val = self.inner.fetch_add(n, Ordering::SeqCst);
35        val < self.size
36    }
37}

39#[derive(Clone)]
40struct Bench {
41    db: Arc<Database>,
42    opt: Opt,
43}
44
45impl Bench {
46    fn run(self) {
47        let opt = &self.opt;
48        let counter = Counter::new(opt.count);
49
50        let mut handles = Vec::new();
51
52        let sw = Stopwatch::start_new();
53
54        let step = opt.queue_depth.div_ceil(opt.threads);
55        let mut start = 0;
56        for _ in 0..opt.threads {
57            let end = std::cmp::min(start + step, opt.queue_depth);
58
59            let range = start..end;
60            let counter = counter.clone();
61            let b = self.clone();
62            let handle = std::thread::spawn(move || {
63                futures::executor::block_on(b.run_range(range, counter))
64            });
65            handles.push(handle);
66
67            start = end;
68        }
69
70        for handle in handles {
71            handle
72                .join()
73                .expect("failed to join")
74                .expect("failed to run bench");
75        }
76
77        let elapsed = sw.elapsed_ms() as usize;
78
79        info!(
80            "bench took: {:?} ms, {:?} tps",
81            elapsed,
82            1000 * opt.count / elapsed
83        );
84    }
85
86    async fn run_range(&self, r: std::ops::Range<usize>, counter: Counter) -> FdbResult<()> {
87        try_join_all(r.map(|n| {
88            // With deterministic Rng, benchmark with same parameters will overwrite same set
89            // of keys again, which makes benchmark result stable.
90            let rng = StepRng::new(n as u64, 1);
91            self.run_bench(rng, counter.clone())
92        }))
93        .await?;
94        Ok(())
95    }
96
97    async fn run_bench(&self, mut rng: StepRng, counter: Counter) -> FdbResult<()> {
98        let mut key_buf = vec![0; self.opt.key_len];
99
100        let mut val_buf = vec![0; self.opt.val_len];
101
102        let trx_batch_size = self.opt.trx_batch_size;
103        let mut trx = self.db.create_trx()?;
104
105        loop {
106            for _ in 0..trx_batch_size {
107                rng.fill_bytes(&mut key_buf);
108                rng.fill_bytes(&mut val_buf);
109                key_buf[0] = 0x01;
110                trx.set(&key_buf, &val_buf);
111            }
112
113            trx = trx.commit().await?.reset();
114
115            if !counter.decr(trx_batch_size) {
116                break Ok(());
117            }
118        }
119    }
120}

123#[structopt(name = "fdb-bench")]

128    #[structopt(short = "q", long = "queue-depth", default_value = "1000")]

131    #[structopt(short = "c", long = "count", default_value = "300000")]

133
134    #[structopt(long = "trx-batch-size", default_value = "10")]
135    trx_batch_size: usize,

137    #[structopt(long = "key-len", default_value = "10")]
138    key_len: usize,

143fn main() {
144    env_logger::init();
