9//! Configuration of foundationDB API and Network
10//!
11//! Provides a safe wrapper around the following C API calls:
12//!
13//! - [API versioning](https://apple.github.io/foundationdb/api-c.html#api-versioning)
14//! - [Network](https://apple.github.io/foundationdb/api-c.html#network)
15//!
16//! The network lifecycle is tracked by a process-global state machine
17//! (`Uninitialized -> ApiVersionSelected -> Running -> Stopped`). Booting is safe
18//! and idempotent, the network is started lazily by [`crate::Database`]
19//! constructors if needed, and it runs until process exit: an atexit hook stops it
20//! and joins the network thread, as required by the C API. Applications that
21//! manage their own shutdown can call [`disable_stop_on_exit`] and/or
22//! [`stop_network`].
23//!
24//! <div class="warning">
25//!
26//! The automatic stop at process exit is meant for tests and short-lived tools.
27//! In a production application, prefer a clean teardown: the network thread is
28//! the event loop driving every transaction and you may still have on-going
29//! operations at exit time. Finish or cancel your work, drop the
30//! [`crate::Database`] handles, then call [`stop_network`] yourself.
31//!
32//! </div>
33

43/// Returns the max api version of the underlying Fdb C API Client
44pub fn get_max_api_version() -> i32 {
45    unsafe { fdb_sys::fdb_get_max_api_version() }
46}

48// FDB C error codes used by the lifecycle state machine
49const ERR_API_VERSION_UNSET: i32 = 2200;
50const ERR_API_VERSION_ALREADY_SET: i32 = 2201;
51const ERR_API_VERSION_NOT_SUPPORTED: i32 = 2203;
52const ERR_NETWORK_ALREADY_SETUP: i32 = 2009;
53const ERR_NETWORK_CANNOT_BE_RESTARTED: i32 = 2025;
54
55/// Process-global lifecycle of the FoundationDB client.
56///
57/// The C API allows selecting the API version and setting up the network once per
58/// process, and the network can never be restarted once stopped. Every FFI call
59/// involved lives in exactly one transition below, serialized by the `NETWORK`
60/// mutex.
61enum NetworkLifecycle {
62    Uninitialized,
63    ApiVersionSelected {
64        api_version: i32,
65    },
66    Running {
67        api_version: i32,
68        handle: JoinHandle<()>,
69    },
70    Stopped {
71        api_version: i32,
72    },
73}
74
75static NETWORK: Mutex<NetworkLifecycle> = Mutex::new(NetworkLifecycle::Uninitialized);
76static STOP_ON_EXIT: AtomicBool = AtomicBool::new(true);
77// Last error returned by fdb_run_network (0 = none). An atomic, not a mutex: the
78// network thread must never take a lock shared with the thread joining it.
79static NETWORK_RUN_ERROR: AtomicI32 = AtomicI32::new(0);
80
81fn lock_network() -> MutexGuard<'static, NetworkLifecycle> {
82    // State is only mutated after the corresponding FFI call succeeded, so a
83    // poisoned lock still guards a consistent state. The one exception is a
84    // thread-spawn panic in start_network after fdb_setup_network succeeded
85    // (OS resource exhaustion): the state then stays ApiVersionSelected while
86    // the network is set up, and any retry fails with 2009. Deliberately not
87    // handled: nothing sensible can be done at that point.
88    NETWORK.lock().unwrap_or_else(PoisonError::into_inner)
89}
90
91impl NetworkLifecycle {
92    fn select_api_version(&mut self, version: i32) -> FdbResult<()> {
93        match self {
94            NetworkLifecycle::Uninitialized => {
95                error::eval(unsafe {
96                    fdb_sys::fdb_select_api_version_impl(version, fdb_sys::FDB_API_VERSION as i32)
97                })
98                .inspect_err(|e| {
99                    // generally means the local libfdb doesn't support the requested version
100                    if e.code() == ERR_API_VERSION_NOT_SUPPORTED {
101                        let max_api_version = get_max_api_version();
102                        if max_api_version < fdb_sys::FDB_API_VERSION as i32 {
103                            eprintln!(
104                                "The version of FoundationDB binding requested '{}' is not supported",
105                                fdb_sys::FDB_API_VERSION
106                            );
107                            eprintln!(
108                                "by the installed FoundationDB C library. Maximum supported version by the local library is {max_api_version}"
109                            );
110                        }
111                    }
112                })?;
113                *self = NetworkLifecycle::ApiVersionSelected {
114                    api_version: version,
115                };
116                Ok(())
117            }
118            NetworkLifecycle::ApiVersionSelected { api_version }
119            | NetworkLifecycle::Running { api_version, .. }
120            | NetworkLifecycle::Stopped { api_version } => {
121                if *api_version == version {
122                    Ok(())
123                } else {
124                    Err(FdbError::from_code(ERR_API_VERSION_ALREADY_SET))
125                }
126            }
127        }
128    }
129
130    fn set_network_option(&mut self, option: NetworkOption) -> FdbResult<()> {
131        match self {
132            // unreachable through the public API: a NetworkBuilder only exists once
133            // an api version is selected
134            NetworkLifecycle::Uninitialized => Err(FdbError::from_code(ERR_API_VERSION_UNSET)),
135            NetworkLifecycle::ApiVersionSelected { .. } => {
136                // SAFETY: the api version is selected and the network is not set up
137                // yet; calls are serialized by the NETWORK mutex.
138                unsafe { option.apply() }
139            }
140            NetworkLifecycle::Running { .. } => Err(FdbError::from_code(ERR_NETWORK_ALREADY_SETUP)),
141            NetworkLifecycle::Stopped { .. } => {
142                Err(FdbError::from_code(ERR_NETWORK_CANNOT_BE_RESTARTED))
143            }
144        }
145    }
146
147    fn start_network(&mut self) -> FdbResult<()> {
148        if matches!(self, NetworkLifecycle::Uninitialized) {
149            self.select_api_version(fdb_sys::FDB_API_VERSION as i32)?;
150        }
151        match self {
152            NetworkLifecycle::Uninitialized => unreachable!("the api version was just selected"),
153            NetworkLifecycle::ApiVersionSelected { api_version } => {
154                let api_version = *api_version;
155                error::eval(unsafe { fdb_sys::fdb_setup_network() })?;
156                // Registered once the network is set up, i.e. after libfdb_c was
157                // loaded: atexit handlers run in LIFO order, so ours stops and joins
158                // the network before any exit-time cleanup of the library itself.
159                if unsafe { libc::atexit(network_atexit_hook) } != 0 {
160                    eprintln!(
161                        "failed to register the FoundationDB atexit hook; the network will not be stopped automatically at process exit"
162                    );
163                }
164                let handle = std::thread::Builder::new()
165                    .name("fdb-network".to_string())
166                    .spawn(run_network_thread)
167                    .expect("failed to spawn the fdb-network thread");
168                *self = NetworkLifecycle::Running {
169                    api_version,
170                    handle,
171                };
172                Ok(())
173            }
174            NetworkLifecycle::Running { .. } => Ok(()),
175            NetworkLifecycle::Stopped { .. } => {
176                Err(FdbError::from_code(ERR_NETWORK_CANNOT_BE_RESTARTED))
177            }
178        }
179    }
180
181    fn stop_network(&mut self) -> FdbResult<()> {
182        match self {
183            NetworkLifecycle::Uninitialized => Ok(()),
184            NetworkLifecycle::ApiVersionSelected { api_version } => {
185                // Nothing is running, but an explicit stop must be terminal so that
186                // a later lazy boot cannot restart the network behind the caller's
187                // back.
188                *self = NetworkLifecycle::Stopped {
189                    api_version: *api_version,
190                };
191                Ok(())
192            }
193            NetworkLifecycle::Running { .. } => {
194                // Stop before taking the variant apart: on error the state stays
195                // Running, handle included.
196                error::eval(unsafe { fdb_sys::fdb_stop_network() })?;
197                let NetworkLifecycle::Running {
198                    api_version,
199                    handle,
200                } = std::mem::replace(self, NetworkLifecycle::Uninitialized)
201                else {
202                    unreachable!("state was just matched as Running");
203                };
204                if handle.join().is_err() {
205                    // run_network_thread cannot panic; keep the exit path panic-free
206                    // anyway (eprintln! panics if the stderr write fails).
207                    let _ = writeln!(std::io::stderr(), "the fdb-network thread panicked");
208                }
209                *self = NetworkLifecycle::Stopped { api_version };
210                Ok(())
211            }
212            NetworkLifecycle::Stopped { .. } => Ok(()),
213        }
214    }
215}
216
217/// Body of the "fdb-network" thread.
218///
219/// Invariant: this function must never lock `NETWORK`, as `stop_network` joins the
220/// thread while holding the lock.
221fn run_network_thread() {
222    if let Err(err) = error::eval(unsafe { fdb_sys::fdb_run_network() }) {
223        NETWORK_RUN_ERROR.store(err.code(), Ordering::Release);
224        let _ = writeln!(
225            std::io::stderr(),
226            "fdb_run_network returned an error: {err}"
227        );
228    }
229}

231/// Stops the network at process exit, as required by the C API contract.
232///
233/// Panic-free: unwinding out of an `extern "C"` function would abort with a less
234/// useful message.
235extern "C" fn network_atexit_hook() {
236    if !STOP_ON_EXIT.load(Ordering::Acquire) {
237        return;
238    }
239    if let Err(err) = lock_network().stop_network() {
240        // Exiting with the network still running is undefined behavior, aborting
241        // is not.
242        let _ = writeln!(
243            std::io::stderr(),
244            "failed to stop the FoundationDB network at exit: {err}"
245        );
246        std::process::abort();
247    }
248}

250/// Stops the FoundationDB network and joins the network thread.
251///
252/// This is terminal for the process: once it returns `Ok`, every FoundationDB
253/// operation fails with error 2025 (`network_cannot_be_restarted`), including in
254/// other threads. It is idempotent. One exception to terminality: if the client
255/// was never initialized at all (no boot, no API version selected, no `Database`
256/// created), the call is a plain no-op and a later boot still succeeds.
257///
258/// Tests and short-lived tools can simply rely on the automatic stop at process
259/// exit. In a production application, prefer calling this yourself as part of a
260/// clean shutdown sequence, once on-going operations are finished and the
261/// [`crate::Database`] handles are dropped.
262///
263/// Must not be called from the network thread itself (e.g. from a future
264/// callback), as it joins that thread.
265///
266/// # Examples
267///
268/// ```rust
269/// foundationdb::boot().expect("failed to initialize FoundationDB");
270/// // ... use FoundationDB ...
271/// foundationdb::api::stop_network().expect("failed to stop the network");
272/// // The stop is terminal: any use of FoundationDB now fails with error 2025.
273/// assert_eq!(foundationdb::boot().err().map(|e| e.code()), Some(2025));
274/// ```
275pub fn stop_network() -> FdbResult<()> {
276    lock_network().stop_network()
277}
278
279/// Disables the automatic network stop at process exit.
280///
281/// The atexit hook registered when the network starts becomes a no-op. Use it
282/// when your application owns its shutdown sequence and calls [`stop_network`]
283/// itself: the C API requires the network to be stopped and joined before a
284/// normal process exit.
285pub fn disable_stop_on_exit() {
286    STOP_ON_EXIT.store(false, Ordering::Release);
287}

289/// Returns the error reported by the network event loop, if it stopped with one.
290pub fn network_run_error() -> Option<FdbError> {
291    match NETWORK_RUN_ERROR.load(Ordering::Acquire) {
292        0 => None,
293        code => Some(FdbError::from_code(code)),
294    }
295}

297/// Starts the network with the default API version if nothing was booted yet.
298///
299/// Lazy-boot entry point used by `Database` constructors.
300pub(crate) fn ensure_network_started() -> FdbResult<()> {
301    lock_network().start_network()

304/// A Builder with which different versions of the Fdb C API can be initialized
305///
306/// ```
307/// foundationdb::api::FdbApiBuilder::default().build().expect("fdb api initialized");
308/// ```
309pub struct FdbApiBuilder {
310    runtime_version: i32,
311}
312
313impl FdbApiBuilder {
314    /// The version of run-time behavior the API is requested to provide.
315    pub fn runtime_version(&self) -> i32 {
316        self.runtime_version
317    }
318
319    /// Set the version of run-time behavior the API is requested to provide.
320    ///
321    /// Must be less than or equal to header_version, `foundationdb_sys::FDB_API_VERSION`, and should almost always be equal.
322    /// Language bindings which themselves expose API versioning will usually pass the version requested by the application.
323    pub fn set_runtime_version(mut self, version: i32) -> Self {
324        self.runtime_version = version;
325        self
326    }
327
328    /// Selects the foundationDB API version and returns a `NetworkBuilder`.
329    ///
330    /// Idempotent: re-selecting the version already in use is a no-op.
331    ///
332    /// # Errors
333    ///
334    /// - error 2201 (`api_version_already_set`) if a different version was selected
335    ///   before
336    /// - error 2203 (`api_version_not_supported`) if the installed libfdb_c does
337    ///   not support the requested version
338    pub fn build(self) -> FdbResult<NetworkBuilder> {
339        lock_network().select_api_version(self.runtime_version)?;
340        Ok(NetworkBuilder { _private: () })
341    }
342}
343
344impl Default for FdbApiBuilder {
345    fn default() -> Self {
346        FdbApiBuilder {
347            runtime_version: fdb_sys::FDB_API_VERSION as i32,
348        }
349    }
350}

352/// A Builder to configure network options and start the network event loop.
353///
354/// ```
355/// use foundationdb::api::FdbApiBuilder;
356///
357/// let network_builder = FdbApiBuilder::default().build().expect("fdb api initialized");
358/// network_builder.boot().expect("fdb network running");
359/// ```
360pub struct NetworkBuilder {
361    _private: (),
362}
363
364impl NetworkBuilder {
365    /// Set network options.
366    ///
367    /// # Errors
368    ///
369    /// - error 2009 (`network_already_setup`) once the network is running
370    /// - error 2025 (`network_cannot_be_restarted`) once the network is stopped
371    pub fn set_option(self, option: NetworkOption) -> FdbResult<Self> {
372        lock_network().set_network_option(option)?;
373        Ok(self)
374    }
375
376    /// Starts the FoundationDB network event loop in a dedicated thread.
377    ///
378    /// Safe and idempotent: booting an already running network is a no-op. The
379    /// network runs until process exit, where an atexit hook stops it and joins the
380    /// network thread, or until an explicit [`stop_network`] call.
381    ///
382    /// # Errors
383    ///
384    /// - error 2025 (`network_cannot_be_restarted`) if the network was stopped
385    ///
386    /// # Examples
387    ///
388    /// ```rust
389    /// use foundationdb::api::FdbApiBuilder;
390    ///
391    /// let network_builder = FdbApiBuilder::default().build().expect("fdb api initialized");
392    /// network_builder.boot().expect("fdb network running");
393    /// // do some interesting things with the API...
394    /// ```
395    ///
396    /// ```rust
397    /// use foundationdb::api::FdbApiBuilder;
398    ///
399    /// #[tokio::main]
400    /// async fn main() {
401    ///     let network_builder = FdbApiBuilder::default().build().expect("fdb api initialized");
402    ///     network_builder.boot().expect("fdb network running");
403    ///     // do some interesting things with the API...
404    /// }
405    /// ```
406    pub fn boot(self) -> FdbResult<NetworkAutoStop> {
407        lock_network().start_network()?;
408        Ok(NetworkAutoStop { _private: () })
409    }
410}
411
412/// Kept for backward compatibility: dropping it does nothing.
413///
414/// The network runs until process exit (an atexit hook stops it and joins the
415/// network thread), or until an explicit [`stop_network`] call.
416pub struct NetworkAutoStop {
417    _private: (),
418}
419
420impl Drop for NetworkAutoStop {
421    fn drop(&mut self) {
422        // Intentionally inert: the network is stopped at process exit (or by an
423        // explicit stop_network()), never on guard drop. Stopping on drop would
424        // make any later use of FoundationDB in the process fail with error 2025.
425    }
426}

428#[cfg(test)]
429mod tests {
430    use super::*;

432    #[test]
433    fn test_max_api() {
434        assert!(get_max_api_version() > 0);
435    }
436
437    // Only exercises the FFI-free arms of the transition table, on locally
438    // constructed states: lib unit tests share one process, so they must neither
439    // touch the NETWORK static nor select a real api version.
440    #[test]
441    fn test_transition_table_ffi_free_arms() {
442        // ApiVersionSelected
443        let mut state = NetworkLifecycle::ApiVersionSelected { api_version: 710 };
444        state
445            .select_api_version(710)
446            .expect("same version must be idempotent");
447        assert_eq!(
448            state.select_api_version(620).unwrap_err().code(),
449            ERR_API_VERSION_ALREADY_SET
450        );
451        state.stop_network().expect("stop before start must be Ok");
452        assert!(matches!(
453            state,
454            NetworkLifecycle::Stopped { api_version: 710 }
455        ));
456
457        // Running (dummy finished thread; stop_network is never called on it)
458        let mut state = NetworkLifecycle::Running {
459            api_version: 710,
460            handle: std::thread::spawn(|| {}),
461        };
462        state
463            .start_network()
464            .expect("start while running must be idempotent");
465        assert_eq!(
466            state.select_api_version(620).unwrap_err().code(),
467            ERR_API_VERSION_ALREADY_SET
468        );
469        assert_eq!(
470            state
471                .set_network_option(NetworkOption::TraceEnable(String::new()))
472                .unwrap_err()
473                .code(),
474            ERR_NETWORK_ALREADY_SETUP
475        );
476
477        // Stopped is terminal
478        let mut state = NetworkLifecycle::Stopped { api_version: 710 };
479        assert_eq!(
480            state.start_network().unwrap_err().code(),
481            ERR_NETWORK_CANNOT_BE_RESTARTED
482        );
483        assert_eq!(
484            state
485                .set_network_option(NetworkOption::TraceEnable(String::new()))
486                .unwrap_err()
487                .code(),
488            ERR_NETWORK_CANNOT_BE_RESTARTED
489        );
490        state.stop_network().expect("stop must be idempotent");
491        state
492            .select_api_version(710)
493            .expect("re-selecting the same version stays Ok once stopped");
494
495        // Uninitialized
496        let mut state = NetworkLifecycle::Uninitialized;
497        state.stop_network().expect("stopping nothing must be Ok");
498        assert!(matches!(state, NetworkLifecycle::Uninitialized));
499        assert_eq!(
500            state
501                .set_network_option(NetworkOption::TraceEnable(String::new()))
502                .unwrap_err()
503                .code(),
504            ERR_API_VERSION_UNSET
505        );
506    }
