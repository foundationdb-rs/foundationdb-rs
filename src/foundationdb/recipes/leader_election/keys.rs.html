8//! Key management utilities for leader election
9//!
10//! This module defines the key structure and namespace organization for
11//! storing leader election data in FoundationDB. All keys are scoped
12//! within a user-provided subspace to allow multiple elections to coexist.
13//!
14//! # Key Schema
15//!
16//! ```text
17//! <subspace>/config           - Election configuration
18//! <subspace>/leader           - Current leader state (single key, O(1) access)
19//! <subspace>/candidates/<id>  - Per-candidate registration and heartbeat
20//! ```
21
22use crate::tuple::Subspace;
23
24/// Key prefixes for different data types
25pub const CONFIG_PREFIX: &str = "config";
26pub const LEADER_PREFIX: &str = "leader";
27pub const CANDIDATES_PREFIX: &str = "candidates";
28
29/// Generate the key for global configuration
30///
31/// Returns the key where election configuration is stored.
32///
33/// # Key Structure
34/// `<subspace>/config`
35pub fn config_key(subspace: &Subspace) -> Vec<u8> {
36    subspace.pack(&(CONFIG_PREFIX,))
37}
38
39/// Generate the key for leader state
40///
41/// Returns the key where current leader information is stored.
42/// This is the core "active disk" key for O(1) leadership operations.
43///
44/// # Key Structure
45/// `<subspace>/leader`
46///
47/// # Contents
48/// Stores the current leader's ballot, ID, priority, lease expiry, and versionstamp
49pub fn leader_key(subspace: &Subspace) -> Vec<u8> {
50    subspace.pack(&(LEADER_PREFIX,))
51}
52
53/// Generate the key for a specific candidate
54///
55/// Returns the key for storing a candidate's registration and heartbeat info.
56///
57/// # Key Structure
58/// `<subspace>/candidates/<process_id>`
59///
60/// # Arguments
61/// * `process_id` - Unique identifier for the candidate process
62pub fn candidate_key(subspace: &Subspace, process_id: &str) -> Vec<u8> {
63    subspace.pack(&(CANDIDATES_PREFIX, process_id))
64}
65
66/// Generate the key range for all candidates
67///
68/// Returns the start and end keys for scanning all registered candidates.
69///
70/// # Key Structure
71/// Range: `<subspace>/candidates/` to `<subspace>/candidates/\xff`
72///
73/// # Usage
74/// Used for listing candidates, evicting dead candidates, or monitoring.
75/// Note: Leadership determination is O(1) and doesn't require scanning.
76pub fn candidates_range(subspace: &Subspace) -> (Vec<u8>, Vec<u8>) {
77    let start = subspace.subspace(&(CANDIDATES_PREFIX,));
78    start.range()
79}
