8//! Ranked register algorithm
9//!
10//! Implements the core read/write/value operations from Chockler & Malkhi's
11//! "Active Disk Paxos with infinitely many processes" (PODC 2002, Section 5.1, Figure 3).
12//!
13//! All functions operate within a FoundationDB transaction for atomicity.
14
15use crate::{
16    tuple::{pack, unpack, Subspace},
17    Transaction,
18};
19use std::ops::Deref;
20
21use super::{errors::Result, keys, types::*};
22
23// ============================================================================
24// STATE HELPERS
25// ============================================================================
26
27/// Read the register state from FoundationDB
28///
29/// Returns `Default` (zero ranks, no value) if the key is absent,
30/// which represents the bottom/uninitialized state.
31async fn read_state<T>(txn: &T, key: &[u8]) -> Result<RegisterState>
32where
33    T: Deref<Target = Transaction>,
34{
35    let data = match txn.get(key, false).await? {
36        Some(d) => d,
37        None => return Ok(RegisterState::default()),
38    };
39
40    // Unpack tuple: (max_read_rank, max_write_rank, has_value, value)
41    let tuple: (u64, u64, bool, Vec<u8>) = unpack(&data)?;
42
43    let value = if tuple.2 { Some(tuple.3) } else { None };
44
45    Ok(RegisterState {
46        max_read_rank: Rank::from(tuple.0),
47        max_write_rank: Rank::from(tuple.1),
48        value,
49    })
50}
51
52/// Write the register state to FoundationDB
53fn write_state<T>(txn: &T, key: &[u8], state: &RegisterState)
54where
55    T: Deref<Target = Transaction>,
56{
57    let has_value = state.value.is_some();
58    let value = state.value.as_deref().unwrap_or(&[]);
59
60    let data = (
61        state.max_read_rank.as_u64(),
62        state.max_write_rank.as_u64(),
63        has_value,
64        value,
65    );
66    let packed = pack(&data);
67    txn.set(key, &packed);
68}

70// ============================================================================
71// CORE OPERATIONS (Paper Section 5.1, Figure 3)
72// ============================================================================
73
74/// Perform a ranked read on the register
75///
76/// Updates `max_read_rank` if the given rank is higher than the current one,
77/// effectively installing a fence that prevents lower-ranked writes.
78/// Returns the current write rank and value.
79///
80/// This is used by the leader before writing to ensure no concurrent
81/// higher-ranked process has written.
82pub async fn read<T>(txn: &T, subspace: &Subspace, rank: Rank) -> Result<ReadResult>
83where
84    T: Deref<Target = Transaction>,
85{
86    let key = keys::state_key(subspace);
87    let mut state = read_state(txn, &key).await?;
88
89    // Update max_read_rank if our rank is higher
90    if rank > state.max_read_rank {
91        state.max_read_rank = rank;
92        write_state(txn, &key, &state);
93    }
94
95    Ok(ReadResult {
96        write_rank: state.max_write_rank,
97        value: state.value,
98    })
99}
100
101/// Perform a ranked write on the register
102///
103/// Commits the value only if:
104/// - `rank >= max_read_rank` (no higher-ranked read has installed a fence)
105/// - `rank > max_write_rank` (no equal-or-higher-ranked write has occurred)
106///
107/// Returns `Committed` if the write succeeds, `Aborted` otherwise.
108pub async fn write<T>(txn: &T, subspace: &Subspace, rank: Rank, value: &[u8]) -> Result<WriteResult>
109where
110    T: Deref<Target = Transaction>,
111{
112    let key = keys::state_key(subspace);
113    let mut state = read_state(txn, &key).await?;
114
115    // Check rank conditions (paper Section 5.1)
116    if rank < state.max_read_rank || rank <= state.max_write_rank {
117        return Ok(WriteResult::Aborted);
118    }
119
120    // Commit the write
121    state.max_write_rank = rank;
122    state.value = Some(value.to_vec());
123    write_state(txn, &key, &state);
124
125    Ok(WriteResult::Committed)
126}
127
128/// Read the current value without updating ranks
129///
130/// This is a plain read for followers/observers. It does NOT update
131/// `max_read_rank`, so it won't interfere with the leader's writes.
132///
133/// Safe to call from any process at any time.
134pub async fn value<T>(txn: &T, subspace: &Subspace) -> Result<Option<Vec<u8>>>
135where
136    T: Deref<Target = Transaction>,
137{
138    let key = keys::state_key(subspace);
139    let state = read_state(txn, &key).await?;
140    Ok(state.value)
141}
