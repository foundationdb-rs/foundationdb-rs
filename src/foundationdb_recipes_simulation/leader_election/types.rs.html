6// Use i64 instead of u8 for tuple packing (u8 is not supported)
7pub(crate) const OP_REGISTER: i64 = 0;
8pub(crate) const OP_HEARTBEAT: i64 = 1;
9pub(crate) const OP_TRY_BECOME_LEADER: i64 = 2;
10pub(crate) const OP_RESIGN: i64 = 3;
11
12/// A single log entry - entries are read in FDB commit order (versionstamp ordering)
13#[derive(Debug, Clone)]
14pub(crate) struct LogEntry {
15    pub versionstamp: Versionstamp,
16    pub client_id: i32,
17    pub op_num: u64,
18    pub op_type: i64,
19    pub success: bool,
20    pub became_leader: bool,
21
22    // Ballot tracking for invariant checks
23    /// Ballot number after the operation completed
24    pub ballot: u64,
25    /// Ballot number before the operation (for detecting transitions)
26    pub previous_ballot: u64,
27
28    // Lease tracking for overlap detection
29    /// Lease expiry timestamp in nanoseconds (for leadership claims)
30    pub lease_expiry_nanos: i64,
31    /// Timestamp when leadership claim was made (for lease validity checks)
32    pub claim_timestamp_nanos: i64,
33}
34
35/// Log entries in FDB commit order (versionstamp-ordered keys give us true ordering)
36pub(crate) type LogEntries = Vec<LogEntry>;
37
38/// Operation types for logging
39#[derive(Clone, Copy, Debug)]
40pub(crate) enum OpType {
41    Register,
42    Heartbeat,
43    TryBecomeLeader,
44    Resign,
45}
46
47impl OpType {
48    pub(crate) fn from_i64(val: i64) -> Option<Self> {
49        match val {
50            OP_REGISTER => Some(OpType::Register),
51            OP_HEARTBEAT => Some(OpType::Heartbeat),
52            OP_TRY_BECOME_LEADER => Some(OpType::TryBecomeLeader),
53            OP_RESIGN => Some(OpType::Resign),
54            _ => None,
55        }
56    }
57
58    pub(crate) fn as_str(self) -> &'static str {
59        match self {
60            OpType::Register => "Register",
61            OpType::Heartbeat => "Heartbeat",
62            OpType::TryBecomeLeader => "TryBecomeLeader",
63            OpType::Resign => "Resign",
64        }
65    }
66}

102/// Clock skew levels for simulation
103#[derive(Clone, Copy, Debug)]
104pub(crate) enum ClockSkewLevel {
105    /// Light: ±100ms (like FDB's timer() vs now())
106    Light,
107    /// Moderate: ±500ms (cloud NTP worst case)
108    Moderate,
109    /// Extreme: ±1s (stress test, will cause election churn)
110    Extreme,
111}
112
113impl ClockSkewLevel {
114    /// Maximum clock offset in seconds for this skew level
115    pub fn max_offset_secs(&self) -> f64 {
116        match self {
117            ClockSkewLevel::Light => 0.1,
118            ClockSkewLevel::Moderate => 0.5,
119            ClockSkewLevel::Extreme => 1.0,
120        }
121    }
