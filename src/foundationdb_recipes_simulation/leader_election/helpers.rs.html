19    /// Helper to get random f64 in [0, 1) using context.rnd()
20    fn rnd_f64(&self) -> f64 {
21        self.context.rnd() as f64 / u32::MAX as f64
22    }
23
24    /// Get local time with simulated clock skew (mimics FDB's timer())
25    ///
26    /// FDB strategy from sim2.actor.cpp:
27    /// timerTime += random01() * (time + 0.1 - timerTime) / 2.0
28    pub(crate) fn local_time(&mut self) -> Duration {
29        let real_time = self.context.now();
30
31        // 1. Apply FDB-style timer drift
32        // Timer moves partway toward (real_time + base_offset + max_drift)
33        let max_ahead = self.clock_skew_level.max_offset_secs();
34        let target = real_time + self.clock_offset_secs + max_ahead;
35        self.clock_timer_time += self.rnd_f64() * (target - self.clock_timer_time) / 2.0;
36
37        // Ensure timer doesn't go backwards (monotonic-ish)
38        self.clock_timer_time = self
39            .clock_timer_time
40            .max(real_time + self.clock_offset_secs);
41
42        // 2. Calculate offset from real time
43        let offset = self.clock_timer_time - real_time;
44
45        // 3. Apply jitter to the OFFSET only (not the entire time)
46        // FDB's DELAY_JITTER applies to delay amounts, not absolute times
47        let jitter_multiplier =
48            DEFAULT_CLOCK_JITTER_OFFSET + DEFAULT_CLOCK_JITTER_RANGE * self.rnd_f64();
49        let jittered_offset = offset * jitter_multiplier;
50
51        let local = real_time + jittered_offset;
52        Duration::from_secs_f64(local.max(0.0))
53    }

120    /// Dump all log entries with running statistics (like AtomicOps dumpLogKV)
121    /// Entries are already in FDB commit order (versionstamp-ordered keys)
122    pub(crate) fn dump_log_entries(&self, entries: &LogEntries) {
123        let mut running_leadership_count: u64 = 0;
124        let mut per_client_ops: BTreeMap<i32, usize> = BTreeMap::new();
125
126        for (idx, entry) in entries.iter().enumerate() {
127            if entry.became_leader {
128                running_leadership_count += 1;
129            }
130            *per_client_ops.entry(entry.client_id).or_default() += 1;
131
132            let op_name = OpType::from_i64(entry.op_type)
133                .map(|o| o.as_str())
134                .unwrap_or("Unknown");
135
136            self.context.trace(
137                Severity::Debug,
138                "LogEntryDump",
139                details![
140                    "Layer" => "Rust",
141                    "Index" => idx,
142                    "Versionstamp" => format!("{:?}", entry.versionstamp),
143                    "ClientId" => entry.client_id,
144                    "OpNum" => entry.op_num,
145                    "OpType" => op_name,
146                    "Success" => entry.success,
147                    "BecameLeader" => entry.became_leader,
148                    "RunningLeadershipCount" => running_leadership_count
149                ],
150            );
151        }
152
153        // Summary per client
154        for (client_id, count) in &per_client_ops {
155            self.context.trace(
156                Severity::Debug,
157                "LogEntryClientSummary",
158                details![
159                    "Layer" => "Rust",
160                    "ClientId" => client_id,
161                    "TotalOps" => count
162                ],
163            );
164        }
165    }

259    /// Extract per-client statistics from log entries
260    /// Entries are already in FDB commit order (versionstamp-ordered keys)
261    pub(crate) fn extract_client_stats(&self, entries: &LogEntries) -> BTreeMap<i32, ClientStats> {
262        let mut stats: BTreeMap<i32, ClientStats> = BTreeMap::new();
263
264        for entry in entries {
265            let client_stats = stats.entry(entry.client_id).or_default();
266            client_stats.op_nums.push(entry.op_num);
267
268            // Count by operation type
269            match OpType::from_i64(entry.op_type) {
270                Some(OpType::Register) => {
271                    client_stats.register_count += 1;
272                    if !entry.success {
273                        client_stats.error_count += 1;
274                    }
275                }
276                Some(OpType::Heartbeat) => {
277                    client_stats.heartbeat_count += 1;
278                    if !entry.success {
279                        client_stats.error_count += 1;
280                    }
281                }
282                Some(OpType::TryBecomeLeader) => {
283                    client_stats.leadership_attempt_count += 1;
284                    if entry.became_leader {
285                        client_stats.leadership_success_count += 1;
286                    }
287                    if !entry.success {
288                        client_stats.error_count += 1;
289                    }
290                }
291                Some(OpType::Resign) => {
292                    if entry.success {
293                        client_stats.resign_count += 1;
294                    }
295                }
296                None => {}
297            }
298        }
299
300        stats
301    }

303    /// Log statistics summary
304    pub(crate) fn log_statistics(&self, entries: &LogEntries, stats: &BTreeMap<i32, ClientStats>) {
305        let total_entries = entries.len();
306        let total_leadership_claims: usize =
307            stats.values().map(|s| s.leadership_success_count).sum();
308        let total_resigns: usize = stats.values().map(|s| s.resign_count).sum();
309        let total_errors: usize = stats.values().map(|s| s.error_count).sum();
310
311        self.context.trace(
312            Severity::Info,
313            "LogStatistics",
314            details![
315                "Layer" => "Rust",
316                "TotalLogEntries" => total_entries,
317                "TotalClients" => stats.len(),
318                "TotalLeadershipClaims" => total_leadership_claims,
319                "TotalResigns" => total_resigns,
320                "TotalErrors" => total_errors
321            ],
322        );
323
324        for (client_id, client_stats) in stats {
325            self.context.trace(
326                Severity::Info,
327                "ClientStatistics",
328                details![
329                    "Layer" => "Rust",
330                    "ClientId" => client_id,
331                    "RegisterCount" => client_stats.register_count,
332                    "HeartbeatCount" => client_stats.heartbeat_count,
333                    "LeadershipAttempts" => client_stats.leadership_attempt_count,
334                    "LeadershipSuccesses" => client_stats.leadership_success_count,
335                    "ResignCount" => client_stats.resign_count,
336                    "Errors" => client_stats.error_count,
337                    "OpCount" => client_stats.op_nums.len()
338                ],
339            );
340        }
341    }
