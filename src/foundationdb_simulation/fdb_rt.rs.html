1//! Asynchronous runtime module
2//!
3//! This module defines the `fdb_spawn` method to run to completion asynchronous tasks containing
4//! FoundationDB futures
5
6#![doc = include_str!("../docs/fdb_rt.md")]
7
8use std::{
9    cell::{RefCell, UnsafeCell},
10    future::Future,
11    mem,
12    pin::Pin,
13    sync::Arc,
14    task::{Context, RawWaker, RawWakerVTable, Waker},
15};
16
17thread_local! {
18    static TASKS: RefCell<Vec<Arc<Task>>> = const { RefCell::new(Vec::new()) };
19}
20
21struct Task {
22    f: UnsafeCell<Option<Pin<Box<dyn Future<Output = ()>>>>>,
23}
24
25impl Task {
26    const VTABLE: RawWakerVTable =
27        RawWakerVTable::new(Self::clone, Self::wake, Self::wake_by_ref, Self::drop);
28
29    fn poll(self: Arc<Self>) {
30        let waker = unsafe { Waker::from_raw(self.clone().into_waker()) };
31        let cx = &mut Context::from_waker(&waker);
32        let slot = unsafe { &mut *self.f.get() };
33        if let Some(mut f) = slot.take() {
34            if f.as_mut().poll(cx).is_pending() {
35                *slot = Some(f);
36            }
37        }
38    }
39
40    fn from_ptr(ptr: *const ()) -> Arc<Self> {
41        unsafe { Arc::from_raw(&*(ptr as *const Self)) }
42    }
43    fn into_waker(self: Arc<Self>) -> RawWaker {
44        RawWaker::new(Arc::into_raw(self) as *const (), &Self::VTABLE)
45    }
46
47    fn clone(ptr: *const ()) -> RawWaker {
48        let task = Self::from_ptr(ptr);
49        mem::forget(task.clone());
50        task.into_waker()
51    }
52    fn wake(ptr: *const ()) {
53        let task = Self::from_ptr(ptr);
54        TASKS.with_borrow_mut(|tasks| tasks.push(task))
55    }
56    fn wake_by_ref(ptr: *const ()) {
57        let task = Self::from_ptr(ptr);
58        TASKS.with_borrow_mut(|tasks| tasks.push(task.clone()));
59        mem::forget(task);
60    }
61    fn drop(ptr: *const ()) {
62        drop(Self::from_ptr(ptr))
63    }
64}
65
66/// Poll all tasks in the pending queue
67pub fn poll_pending_tasks() {
68    let mut tasks = TASKS.with_borrow_mut(mem::take);
69    tasks.drain(..).for_each(Task::poll);
70}
71
72/// Spawn an async block and resolve all contained FoundationDB futures
73pub(crate) fn fdb_spawn<F>(future: F)
74where
75    F: Future<Output = ()> + 'static,
76{
77    let task = Arc::new(Task {
78        f: UnsafeCell::new(Some(Box::pin(future))),
79    });
80    task.poll();
81}
