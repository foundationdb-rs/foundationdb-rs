19// -----------------------------------------------------------------------------
20// User friendly types
21
22/// Rust representation of a simulated FoundationDB database
23pub type SimDatabase = ManuallyDrop<Database>;
24/// Rust representation of a FoundationDB workload
25pub type Workload = Box<dyn RustWorkload>;
26
27/// RustWorkload trait provides a one to one equivalent to the C++ abstract class `FDBWorkload`
28pub trait RustWorkload {
29    /// Return the name or description of the workload.
30    /// Primarily used for tracing.
31    fn description(&self) -> String;
32
33    /// This method is called by the tester during the setup phase.
34    /// It should be used to populate the database.
35    ///
36    /// # Arguments
37    ///
38    /// * `db` - The simulated database.
39    /// * `done` - A promise that should be resolved to indicate completion
40    fn setup(&'static mut self, db: SimDatabase, done: Promise);
41
42    /// This method should run the actual test.
43    ///
44    /// # Arguments
45    ///
46    /// * `db` - The simulated database.
47    /// * `done` - A promise that should be resolved to indicate completion
48    fn start(&'static mut self, db: SimDatabase, done: Promise);
49
50    /// This method is called when the tester completes.
51    /// A workload should run any consistency/correctness tests during this phase.
52    ///
53    /// # Arguments
54    ///
55    /// * `db` - The simulated database.
56    /// * `done` - A promise that should be resolved to indicate completion
57    fn check(&'static mut self, db: SimDatabase, done: Promise);
58
59    /// If a workload collects metrics (like latencies or throughput numbers), these should be reported back here.
60    /// The multitester (or test orchestrator) will collect all metrics from all test clients and it will aggregate them.
61    fn get_metrics(&self) -> Vec<Metric>;
62
63    /// Set the check timeout for this workload.
64    fn get_check_timeout(&self) -> f64;
65}

74// -----------------------------------------------------------------------------
75// C++ to Rust bindings
76
77#[no_mangle]
78extern "C" fn workload_instantiate(
79    raw_name: *const c_char,
80    raw_context: *mut opaque::Context,
81) -> *mut Workload {
82    let name = str_from_c(raw_name);
83    let context = WorkloadContext::new(raw_context);
84    let workload = unsafe { workload_instantiate_hook(&name, context) };
85    // the `Box<dyn RustWorkload>` is put on the heap with another `Box::new`
86    // `Box::into_raw` turns that `Box` into a thin pointer
87    // it is this pointer that will be stored in the C++ `WorkloadTranslater`
88    // and that is passed to the other `workload_*` functions as `&'static Workload` or `&Workload`
89    // `Box::from_raw` will be called by `workload_drop` to clean up everything
90    Box::into_raw(Box::new(workload))
91}
92#[no_mangle]
93extern "C" fn workload_description(workload: &Workload) -> *const c_char {
94    let description = str_for_c(workload.description());
95    // FIXME: the CString will be dropped by Rust before it is read by the C++ side
96    // but if Rust doesn't drop it now it's a memory leak...
97    // note that that C++ instantly makes a copy so the pointer doesn't stay dangling too long
98    description.as_ptr()
99}
100#[no_mangle]
101extern "C" fn workload_setup(
102    workload: &'static mut Workload,
103    raw_database: NonNull<FDBDatabase>,
104    raw_promise: *const opaque::Promise,
105) {
106    let db = ManuallyDrop::new(Database::new_from_pointer(raw_database));
107    let done = Promise::new(raw_promise);
108    workload.setup(db, done);
109}

130#[no_mangle]
131extern "C" fn workload_get_metrics(workload: &Workload, out: *const opaque::Metrics) {
132    let metrics = workload.get_metrics();
