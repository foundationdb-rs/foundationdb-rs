5/// A structure that represents a block of data associated with a specific transaction.
6/// This structure provides functionality to manage and track progress as data chunks
7/// are added until a target chunk count is reached.
8///
9/// # Overview
10/// `RawTransactionProfilingBlock` is designed to aggregate data chunks that belong to the same transaction.
11/// It keeps track of the progress and allows checking when all chunks are accumulated.
12///
13/// # Fields
14/// - `transaction_id`: A unique identifier for the transaction.
15/// - `buffer`: A vector of bytes that stores the accumulated data chunks.
16/// - `target_chunks`: The total number of chunks expected to complete the data block.
17/// - `accumulated_chunks`: The number of chunks added so far.
18///
19/// # Usage
20/// Use the methods provided by `RawTransactionProfilingBlock` to manipulate its data:
21/// - Create a new instance using the `new` method.
22/// - Add data chunks using the `add_chunk` method.
23/// - Verify if the data block is complete with the `is_complete` method.
24/// - Retrieve the collected data using the `get_data` method.
25///
26/// # Example
27/// ```ignore
28///
29/// let transaction_id = TransactionId::new();
30/// let mut data_block = RawTransactionProfilingBlock::new(transaction_id, 3);
31///
32/// data_block.add_chunk(b"chunk1");
33/// data_block.add_chunk(b"chunk2");
34/// data_block.add_chunk(b"chunk3");
35///
36/// assert!(data_block.is_complete());
37/// assert_eq!(data_block.get_data(), b"chunk1chunk2chunk3");
38/// ```
39pub struct RawTransactionProfilingBlock {
40    transaction_id: TransactionId,
41    buffer: Vec<u8>,
42    target_chunks: usize,
43    accumulated_chunks: usize,
44}
45
46impl Debug for RawTransactionProfilingBlock {
47    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
48        let buffer = Bytes::from(&self.buffer[..]);
49
50        f.debug_struct("RawTransactionProfilingBlock")
51            .field("transaction_id", &self.transaction_id)
52            .field(
53                "progress",
54                &format!("{}/{}", self.accumulated_chunks, self.target_chunks),
55            )
56            .field("buffer", &buffer)
57            .finish()
58    }
59}
60
61impl RawTransactionProfilingBlock {
62    pub fn new(transaction_id: TransactionId, target_chunks: usize) -> Self {
63        Self {
64            transaction_id,
65            buffer: vec![],
66            target_chunks,
67            accumulated_chunks: 0,
68        }
69    }
70
71    pub fn add_chunk(&mut self, chunk: &[u8]) {
72        self.buffer.extend_from_slice(chunk);
73        self.accumulated_chunks += 1;
74    }
75
76    pub fn is_complete(&self) -> bool {
77        self.accumulated_chunks == self.target_chunks
78    }
79
80    pub fn get_data(&self) -> &[u8] {
81        &self.buffer
82    }
83}
