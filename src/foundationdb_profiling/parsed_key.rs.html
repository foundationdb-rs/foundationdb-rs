9/// Define the number of bytes taken by the '/'
10/// separator, so 1 byte
11const SEPARATOR_BYTES_SIZE: usize = 1;
12
13/// VersionStamp is defined as Big endian 10-byte integer.
14/// First/high 8 bytes are a database version, next two are batch version.
15/// For a total of 10 bytes
16const VERSIONSTAMP_BYTES_SIZE: usize = 10;
17
18/// Transaction ID is defined as 16 bytes identifier
19const TRANSACTION_ID_BYTES_SIZE: usize = 16;
20
21/// Chunk number or total chunk number are
22/// defined as BigEndian 4 bytes integer
23const CHUNK_FIELD_SIZE: usize = 4;
24
25/// The `TransactionId` struct encapsulates the unique 16-byte identifier for a
26/// transaction used in FoundationDB's profiling system.
27///
28/// This identifier is part of the database key structure and is used to group
29/// events associated with a specific transaction. The structure is parsed from
30/// the raw key data as part of the profiling analyzer.
31///
32/// # Internal Representation
33/// Internally, this struct wraps a `Vec<u8>` which contains exactly 16 bytes.
34/// The fixed-size representation ensures compatibility with FoundationDB's
35/// schema for profiling-related keys.
36pub struct TransactionId(Vec<u8>);
37
38impl Debug for TransactionId {
39    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
40        let transaction_id = Bytes::from(&self.0[..]);
41        write!(f, "{transaction_id}")
42    }
43}
44
45#[async_trait::async_trait]
46impl Parse for TransactionId {
47    async fn parse(scanner: &mut Scanner<'_>) -> Result<Self, Box<dyn Error + Send + Sync>> {
48        let mut buf = [0_u8; TRANSACTION_ID_BYTES_SIZE];
49        scanner.read_exact(&mut buf).await?;
50        Ok(TransactionId(buf.to_vec()))
51    }
52}

54/// A struct representing a chunk field of a parsed key.
55/// Can be either a total chunk number or a chunk number
56struct ChunkField(usize);
57
58#[async_trait::async_trait]
59impl Parse for ChunkField {
60    async fn parse(scanner: &mut Scanner<'_>) -> Result<Self, Box<dyn Error + Send + Sync>> {
61        let mut buf = [0_u8; CHUNK_FIELD_SIZE];
62        scanner.read_exact(&mut buf).await?;
63        let data = u32::from_be_bytes(buf) as usize;
64        Ok(ChunkField(data))
65    }
66}

68/// A struct that represents a logical version, encapsulating a `Versionstamp`.
69struct Version(Versionstamp);
70
71impl Debug for Version {
72    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
73        let version = self.0.transaction_version()[..8].to_vec();
74        let version = u64::from_be_bytes(version.try_into().map_err(|_err| std::fmt::Error)?);
75        write!(f, "{version}")
76    }
77}
78
79#[async_trait::async_trait]
80impl Parse for Version {
81    async fn parse(scanner: &mut Scanner<'_>) -> Result<Self, Box<dyn Error + Send + Sync>> {
82        let mut buffer = [0_u8; VERSIONSTAMP_BYTES_SIZE];
83        scanner.read_exact(&mut buffer).await?;
84        Ok(Version(Versionstamp::complete(buffer, 0)))
85    }
86}

88/// ProfilingParsedKey represents a structured result of a key parsed from the database.
89///
90/// # Fields
91/// - `transaction_id`: The unique identifier for the transaction.
92/// - `version`: The version of the database associated with the key.
93/// - `current_chunk`: The current chunk index part of the parsed key.
94/// - `total_chunk`: The total number of chunks related to the key.
95pub struct ProfilingParsedKey {
96    pub transaction_id: TransactionId,
97    version: Version,
98    current_chunk: usize,
99    pub total_chunk: usize,
100}
101
102impl ProfilingParsedKey {
103    fn new(
104        version: Version,
105        transaction_id: TransactionId,
106        current_chunk: usize,
107        total_chunk: usize,
108    ) -> Self {
109        ProfilingParsedKey {
110            version,
111            transaction_id,
112            current_chunk,
113            total_chunk,
114        }
115    }
116}
117
118impl Debug for ProfilingParsedKey {
119    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
120        f.debug_struct("ParsedKey")
121            .field("version", &self.version)
122            .field("transaction_id", &self.transaction_id)
123            .field("chunk_id", &self.current_chunk)
124            .field("chunk_total", &self.total_chunk)
125            .finish()
126    }
127}

129/// Parses a database key into a `ParsedKey` structure.
130///
131/// # Arguments
132///
133/// - `key`: The raw byte slice representing the database key.
134///
135/// # Returns
136///
137/// A `Result` containing a `ParsedKey` if parsing is successful, or an error
138/// if the parsing fails.
139///
140/// The function expects the key to follow the specific format:
141///    - A prefix (`PROFILE_PREFIX`) which is skipped.
142///    - A `Version` field parsed from the key.
143///    - A separator (`/`), accounted for by a single byte (defined by `SEPARATOR_BYTES_SIZE`).
144///    - A `TransactionId` field.
145///    - Another separator (`/`).
146///    - Two `ChunkField`s representing chunk number and total chunks.
147///    - A final separator (`/`).
148///
149/// Parsing converts this data into a `ParsedKey` structure which represents
150/// the fields and their respective values in a structured format.
151///
152/// # Errors
153///
154/// This function can return errors in cases such as:
155/// - Invalid format of the provided key.
156/// - Failure to read the key's fields due to incorrect key length or content.
157pub async fn parse_key(key: &[u8]) -> Result<ProfilingParsedKey, Box<dyn Error + Send + Sync>> {
158    let mut scanner = Scanner::new(key);
159    scanner.bump_by(PROFILE_PREFIX.len());
160    let version = Version::parse(&mut scanner).await?;
161    scanner.bump_by(SEPARATOR_BYTES_SIZE);
162    let transaction_id = TransactionId::parse(&mut scanner).await?;
163    scanner.bump_by(SEPARATOR_BYTES_SIZE);
164    let chunk_number = ChunkField::parse(&mut scanner).await?.0;
165    let chunk_total = ChunkField::parse(&mut scanner).await?.0;
166    scanner.bump_by(SEPARATOR_BYTES_SIZE);
167
168    let parsed_key = ProfilingParsedKey::new(version, transaction_id, chunk_number, chunk_total);
169
170    Ok(parsed_key)
171}

173#[cfg(test)]
174mod tests {
175    use super::*;
176    #[tokio::test]
177    async fn test_parse_key() {
178        let key = b"\xff\x02/fdbClientInfo/client_latency/\x19\x5a\x63\xa7\x00\x2f\x3b\xa1\xd8\x00/\x5a\x0c\xd7\xad\x1f\x66\x2c\xb4\xa6\x72\xdb\x45\x3a\x68\xf5\xc1/\x00\x00\x00\x01\x00\x00\x00\x03";
179        let parsed_key = parse_key(key).await.expect("Unable to parse key");
180        assert_eq!(
181            parsed_key.transaction_id.0,
182            [
183                0x5a, 0x0c, 0xd7, 0xad, 0x1f, 0x66, 0x2c, 0xb4, 0xa6, 0x72, 0xdb, 0x45, 0x3a, 0x68,
184                0xf5, 0xc1
185            ]
186        );
187        assert_eq!(
188            parsed_key.version.0.transaction_version(),
189            b"\x19\x5a\x63\xa7\x00\x2f\x3b\xa1\xd8\x00"
190        );
191        assert_eq!(parsed_key.current_chunk, 1);
192        assert_eq!(parsed_key.total_chunk, 3);
193    }
