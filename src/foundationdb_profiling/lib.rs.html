1//! FoundationDB natively implements transaction profiling and analyzing.
2//!
3//! The transactions are sampled at the specified rate and all the events for that sampled transaction are recorded.
4//! Then at 30 second interval, the data for all the sampled transactions during that interval is flushed to the database.
5//! The sampled data is written into special key space `\xff\x02/fdbClientInfo/ - \xff\x02/fdbClientInfo0`
6//!
7//! [source](https://apple.github.io/foundationdb/transaction-profiler-analyzer.html)
8//!
9//! Each data are recorded as chunked events. Events are referenced by the tuple (VersionStamp, TransactionId),
10//! then a header defined how much chunks defined the data block.
11//!
12//! Profiling Keys look like this:
13//! ```ignore
14//!     FF               - 2 bytes \xff\x02
15//!     SSSSSSSSSS       - 10 bytes Version Stamp
16//!     RRRRRRRRRRRRRRRR - 16 bytes Transaction id
17//!     NNNN             - 4 Bytes Chunk number
18//!     TTTT             - 4 Bytes Total number of chunks
19//!```
20//! [source](https://github.com/apple/foundationdb/blob/main/contrib/transaction_profiling_analyzer/transaction_profiling_analyzer.py#L413)
21//!
22//! To get the data block, each chunk which composes the data block must be accumulated until
23//! reaching the "chunk number"
24//!
25//! Noted: It could have more than one event in the data block
26

39/// Defined by the following [pattern](https://github.com/apple/foundationdb/blob/main/contrib/transaction_profiling_analyzer/transaction_profiling_analyzer.py#L421)
40pub const PROFILE_PREFIX: &[u8; 32] = b"\xff\x02/fdbClientInfo/client_latency/";
41
42/// Retrieves raw profiling datablocks stored in the database within the specified read version range.
43///
44/// This function fetches raw events from the FoundationDB range key-value store. The keys are
45/// processed to parse the profiling event data. Events are grouped into chunks and consolidated
46/// into complete [RawTransactionProfilingBlock] instances. The resulting stream yields these complete profiling event
47/// data blocks. Events are referenced using a key structure which includes the transaction ID,
48/// version stamp, and a chunk identifier.
49///
50/// # Arguments
51///
52/// * `trx` - A reference to the FoundationDB `Transaction` used for fetching data.
53/// * `start_read_version` - An optional lower bound read version. If provided, only events
54///   occurring at or after this version are fetched.
55/// * `end_read_version` - An optional upper bound read version. If provided, only events
56///   occurring before this version are fetched.
57///
58/// # Returns
59///
60/// Returns an asynchronous stream (`impl Stream`) where each item of the stream is either:
61/// * A `Result<DataBlock, FdbBindingError>`
62///   - `DataBlock`: A complete profiling event data block.
63///   - `FdbBindingError`: An error that occurred during data retrieval or processing.
64///
65/// # Errors
66///
67/// This function will return an error in the following cases:
68/// * If the transaction (provided by `trx`) encounters issues fetching ranges.
69/// * If an error occurs during key or value parsing.
70///
71/// # Notes
72///
73/// Profiling keys follow a specific binary layout defined by the FoundationDB profiling data space:
74/// * `\xff\x02/fdbClientInfo/client_latency/` acts as the prefix for profiling events.
75/// * Events are referred to by tuple: (VersionStamp, TransactionId).
76/// * Events contain chunks, defined by a chunk number and total chunks.
77///
78/// The function ensures fully constructed events are yielded only once all their chunks are retrieved.
79///
80/// # Example
81///
82/// ```no_run
83/// use foundationdb::{Database, Transaction};
84/// use futures_util::StreamExt;
85/// use foundationdb_profiling::get_raw_datablocks;
86///
87/// async fn example_usage(trx: &Transaction) {
88///     let mut stream = get_raw_datablocks(trx, Some(1), Some(10));
89///     while let Some(event) = stream.next().await {
90///         match event {
91///             Ok(data_block) => {
92///                 println!("Received a data block: {:?}", data_block);
93///             }
94///             Err(e) => {
95///                 eprintln!("Error: {:?}", e);
96///             }
97///         }
98///     }
99/// }
100/// ```
101pub async fn get_raw_datablocks(
102    trx: &Transaction,
103    start_read_version: Option<u64>,
104    end_read_version: Option<u64>,
105) -> impl Stream<Item = Result<RawTransactionProfilingBlock, FdbBindingError>> + use<'_> {
106    // build the start prefix of the range
107    let start_key = if let Some(start_version) = start_read_version {
108        let mut key = PROFILE_PREFIX.to_vec();
109        key.extend_from_slice(&start_version.to_be_bytes());
110        // the start_version is a read version on 8 bytes
111        // to build the VersionStamp the batch version must be added
112        // as we don't know it, we append 2 0x0 bytes
113        // https://github.com/apple/foundationdb/blob/main/contrib/transaction_profiling_analyzer/transaction_profiling_analyzer.py#L463
114        key.extend_from_slice(b"\x00\x00");
115        key
116    } else {
117        // If there is no start_version, then the scan begins
118        // at the beginning of the Profiling Keyspace
119        PROFILE_PREFIX.to_vec()
120    };
121
122    let end_key = if let Some(end_version) = end_read_version {
123        let mut key = PROFILE_PREFIX.to_vec();
124        key.extend_from_slice(&end_version.to_be_bytes());
125        key.extend_from_slice(b"\x00\x00");
126        key
127    } else {
128        // If there is no end_version, then the scan ends
129        // at the very last key of the Profiling Keyspace
130        let mut key = PROFILE_PREFIX.to_vec();
131        key.extend_from_slice(b"\xff");
132        key
133    };
134
135    // Create an async iterator which yields completed DataBlock
136    async_stream::try_stream! {
137        // Get an iterator over the range inside the Profiling Keyspace
138        let profiling_events_raw_key_value_stream = trx
139        .get_ranges_keyvalues(RangeOption::from((start_key, end_key)), true);
140
141        // stream must be pinned to be moveable out of the function
142        let mut profiling_events_raw_key_value_stream = pin!(profiling_events_raw_key_value_stream);
143
144        // Keeps track of the actual Datablock in progress
145        let mut in_progress_datablock = None;
146
147        loop {
148            // Get a key/value
149            let raw_key_value = profiling_events_raw_key_value_stream
150                .try_next()
151                .await
152                .map_err(FdbBindingError::from)?;
153
154            // Take decision on raw data
155            match raw_key_value {
156                // end of the range, no more raw keys
157                None => {
158                    break
159                }
160                // available raw key
161                Some(data) => {
162                    // Try to parse the raw data as a ParsedKey
163                    let parsed_key = parse_key(data.key())
164                        .await
165                        .map_err(FdbBindingError::new_custom_error)?;
166
167                    // Take decision on what doing with the ParsedKey
168                    match in_progress_datablock {
169                        // No DataBlock in progress
170                        None => {
171                            // Create a new Datablock from current ParsedKey
172                            let mut datablock = RawTransactionProfilingBlock::new(parsed_key.transaction_id, parsed_key.total_chunk);
173                            // Accumulate the first data chunk
174                            datablock.add_chunk(data.value());
175                            // If the DataBlock is whole
176                            if datablock.is_complete() {
177                                // yield a DataBlock
178                                yield datablock
179                            } else {
180                                // Set the DataBlock as new in progress to
181                                // accumulate more data chunks
182                                in_progress_datablock = Some(datablock)
183                            }
184
185                        }
186                        // If DataBlock is progressing
187                        Some(ref mut event) => {
188                            // Add the new data chunk
189                            event.add_chunk(data.value());
190                            // Yield DataBlock if complete
191                            if event.is_complete() {
192                                // 'take()' reset the `in_progress_datablock` state
193                                if let Some(event) = in_progress_datablock.take() {
194                                    yield event
195                                }
196                            }
197                        }
198                    }
199                }
200            }
201        }
202    }
203}
