macro_rules! details {
    ($($k:expr => $v:expr),* $(,)?) => { ... };
}
Expand description

Macro that can be used to create Details more easily.

let details1 = vec![
    ("key1".into(), "val1".into()),
    ("key2".into(), format!("key{}", 2)),
];
let details2 = details[
    "key1" => "val1",
    "key2" => format!("key{}", 2),
];
assert_eq!(details1, details2);
